import { fetcher } from '@shared/api/httpClient';
import { User } from '../model/user';

/**
 * Fetches a list of users from the randomuser.me API.
 *
 * @param {number} results - Number of users to fetch
 * @returns {Promise<User[]>} - Promise resolving to a list of users
 */

const USER_API_URL = 'https://randomuser.me/api';

export const getUsers = async (results = 20): Promise<User[]> => {
  const data = await fetcher(`${USER_API_URL}/?results=${results}`);
  return data.results;
};
